## Grouped Bar chart show how the valuation and funding of companies vary across the top 10 industries(based by Valuation)

# Group the data by industry and compute the mean valuation and funding for each group
grouped_uc = uc.groupby('Industry').agg({'Valuation': 'mean', 'Funding': 'mean'})

# Get the top 10 industries based on mean valuation
top_industries = grouped_uc.nlargest(10, 'Valuation')

# Get the list of industries and the number of industries
industries = top_industries.index.tolist()
num_industries = len(industries)

# Get the mean valuation and funding for each industry
mean_valuation = top_industries['Valuation'].tolist()
mean_funding = top_industries['Funding'].tolist()

# Set the width of each bar
bar_width = 0.45

# Set the positions of the x-axis tick labels
tick_positions = np.arange(num_industries)

# Create a figure and axis object with larger size
fig, ax = plt.subplots(figsize=(12, 8))

# Create the first set of bars for mean valuation
ax.bar(tick_positions, mean_valuation, bar_width, label='Mean Valuation')

# Create the second set of bars for mean funding
ax.bar(tick_positions + bar_width, mean_funding, bar_width, label='Mean Funding')

# Add labels, title, and legend to the chart
ax.set_xlabel('Industry')
ax.set_ylabel('Amount (in billions $)')
ax.set_title('Mean Valuation and Funding by Top 10 Industries')
ax.set_xticks(tick_positions + bar_width / 2)
ax.set_xticklabels(industries)

# Set x-axis tick labels with rotation
ax.set_xticklabels(industries, rotation=45, ha='right')

ax.legend()

# Show the chart
plt.show()

## Using stacked Bar chart to visualise the industry and country most investors are investing in the past 2 years

# Filter the data to only include rows with founded year in 2020 or 2021
filtered_uc = uc[(uc['FoundedYr'] == 2020) | (uc['FoundedYr'] == 2021)]

# Group the filtered data by industry, country, and founded year
grouped_uc = filtered_uc.groupby(['Industry', 'Country', 'FoundedYr']).size().unstack(fill_value=0)

# Create the stacked bar chart
grouped_uc.plot(kind='bar', stacked=True, figsize=(12,8))

# Set the title of the chart
plt.title('Relationship between Industry, Country, and Recent Years)')

# Set the x-axis and y-axis labels
plt.xlabel('Industry and Country')
plt.ylabel('Count')

# Show the chart
plt.show()
